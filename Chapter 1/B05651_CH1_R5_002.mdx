WITH
MEMBER [Measures].[Yearly Income] AS
   [Customer].[Customer Geography].currentmember
     .Properties("Yearly Income")

SELECT
   { [Measures].[Internet Sales Amount],
     [Measures].[Yearly Income]
   } on 0,
   TopCount(
      [Customer].[Customer Geography].[Customer].MEMBERS,
      10,
      [Measures].[Internet Sales Amount]
      ) ON 1
FROM
   [Adventure Works]

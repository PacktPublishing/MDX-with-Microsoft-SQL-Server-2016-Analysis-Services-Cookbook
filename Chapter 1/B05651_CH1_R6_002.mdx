    [Measures].[Internet Sales Amount] on 0,
    TopCount (
        [Product].[Subcategory].children,
        5,
        [Measures].[Internet Sales Amount] 
    ) ON 1
FROM
   [Adventure Works]
WHERE
   ( [Date].[Calendar].[Calendar Quarter].&[2013]&[1],
     [Sales Territory].[Sales Territory Region].[Southwest]
   )

Scope( [Time Calcs].[Calc].[Prev period %] );
    This = iif( [Time Calcs].[Calc].[Prev period] = 0, null,
                [Time Calcs].[Calc].&[0] /
                [Time Calcs].[Calc].[Prev period] - 1);
    Format_String(This) = 'Percent';
    Fore_Color(This) = Abs( Measures.CurrentMember < 0 ) * 255;
End Scope;

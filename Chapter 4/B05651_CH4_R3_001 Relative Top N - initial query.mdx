WITH
SET [Ordered Resellers] AS
	Order( [Reseller].[Reseller].[Reseller].MEMBERS,
	       [Measures].[Reseller Sales Amount] ,
	       BDESC )
SELECT
	{ [Measures].[Reseller Sales Amount] } ON 0,
	NON EMPTY
	{ [Sales Territory].[Sales Territory Country].MEMBERS *
	  [Ordered Resellers] } ON 1
FROM
	[Adventure Works]
